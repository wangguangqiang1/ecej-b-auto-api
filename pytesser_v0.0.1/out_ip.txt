function modifyAdmin() {

		$('#modifyPasswordForm').bindVerify();

		var id =$('#loginIdTOP').val();

		var oldPassword = $('#oldPasswordTOP').val() == null ? "" : $.trim($(
				'#oldPasswordTOP').val());
		var password = $('#newPasswordTOP').val() == null ? "" : $.trim($(
				'#newPasswordTOP').val());
		var password2 = $('#newPassword2TOP').val() == null ? "" : $.trim($(
				'#newPassword2TOP').val());
		if (password != password2) {
			bootbox.alert('两次输入的密码不一致，请重新输入');
			return false;
		}

		if ($('#modifyPasswordForm').validates()) {
			$.ajax({
				url : "/userManage/resetPassword?m=Math.random()",
				data : {
					id : id,
					name : name,
					password : password,
					oldPassword:oldPassword
				},
				type : "post",
				dataType : "JSON",
				async : false,
				success : function(data) {
					if (data.success) {
						$('#modifyPasswordDialog').modal('hide');

						bootbox.alert({
							 message: '修改成功',
					            callback: function() {
					            	window.location.href="/login/toLoginPage";
					            }
						});


					} else {
						if (data.flag==1) {
							bootbox.alert("重置失败，原密码输入错误");
							return false;
						}
						bootbox.alert("重置失败，请重新操作");
					}
				}
			});
	}
}
